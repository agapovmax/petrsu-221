b. Установите связи между таблицами в соответствии с внешними ключами. Для каждой

Create Table tblTeacher(	--учитель, справочная
intTeacherId integer identity(1,1),
intTeacherName nvarchar(150) not Null,
intTeacherYear integer not Null,
fltTeacherSalary decimal(10,2) check (fltTeacherSalary >= 0),
Primary key (intTeacherId));

Create Table tblSubject(	--предмет, справочная
intSubjectId integer identity(1,1),
intSubjectName nvarchar(20) not Null,
intSubjectVolume integer default 0,
intTeacherId integer not Null,
intLessonCount integer default 0,
Primary key (intSubjectId),
Foreign key (intTeacherId) references tblTeacher(intTeacherId) on update cascade on delete no action);

Create Table tblLesson(		--урок, операционная
intLessonId integer identity(1,1),
intSubjectId integer not Null,
darLessonDate date not Null,
txtTheme nvarchar(100),
Primary key (intLessonId),
Foreign key (intSubjectId) references tblSubject(intSubjectId) on update cascade on delete no action);

Create Table tblPupil(		--ученик, операционная
intPupilId integer identity(1,1),
txtPupilSurname nvarchar(30) not Null,
txtPupilName nvarchar(25) not Null,
datBirthday date not Null,
chrPupilSex nvarchar(1) default 'м',
txtAddress nvarchar(255),
Primary key (intPupilId));

Create Table tblMark(		--оценка, операционная
intMarkId integer identity(1,1),
intLessonId integer not Null,
intPupilId integer not Null,
intMarkValue integer not Null,
txtMarkComment nvarchar(200),
Primary key (intMarkId),
Foreign key (intLessonId) references tblLesson(intLessonId) on update cascade on delete no action,
Foreign key (intPupilId) references tblPupil(intPupilId) on update cascade on delete no action);

WITH RECURSIVE cte AS (
    SELECT 
        DATE '2024-09-02' AS lesson_date,
        (RANDOM() * 9 + 1)::INTEGER AS subject_id,
        ROW_NUMBER() OVER() AS rn
    UNION ALL
    SELECT 
        lesson_date + INTERVAL '7 days',
        (RANDOM() * 9 + 1)::INTEGER,
        rn + 1
    FROM cte
    WHERE rn < 30
)

--задание 2. Для каждого учителя посчитать общее количество проведенных уроков и количество выставленных оценок. Вывести имя учителя и оба посчитанных значения.
* не использовать alias. В сложных запросах, сложно разобраться
SELECT 
    t.intTeacherName,	-- ищем ФИО учителей
    COUNT(DISTINCT l.intLessonId) AS LessonsCount, --считаем только уникальные значения уроков чтобы не считать
    COUNT(m.intMarkId) AS MarksCount	--считаем оценки
FROM tblTeacher t
JOIN tblSubject s ON t.intTeacherId = s.intTeacherId
JOIN tblLesson l ON s.intSubjectId = l.intSubjectId
LEFT JOIN tblMark m ON l.intLessonId = m.intLessonId
GROUP BY t.intTeacherName;

--задание 3. Успеваемость по конкретному предмету. Вывести список всех уроков по предмету "математика", и для каждого посчитать среднюю оценку. Показать только те уроки, где средняя оценка ниже 4.0. Отсортировать результаты по возрастанию средней оценки.
SELECT 
    --l.intLessonId, --l - это алиас таблицы внутри запроса. Какой собственно урок
    l.txtTheme,	--тема урока
    AVG(m.intMarkValue) AS AverageMark --среднее по оценкам выставленным
FROM tblLesson l --из таблицы уроков
JOIN tblMark m ON l.intLessonId = m.intLessonId --соединяем таблицу уроков с таблицей оценок. Это даст для каждого урока (l) получить все связанные с ним оценки (m)
JOIN tblSubject s ON l.intSubjectId = s.intSubjectId --теперь соединяем таблицу уроков с таблицей предметов. Это нужно, чтобы отфильтровать уроки по названию предмета
WHERE s.intSubjectName = 'Математика' --ищем математику для группировки её оценок
GROUP BY l.intLessonId, l.txtTheme --да, нужно указать все таблицы из селекта вначале, чтобы можно было получить группировку
HAVING AVG(m.intMarkValue) < 4.0 --ищем только оценки ниже 4
ORDER BY AverageMark ASC; --сортируем

--задание 4. Стоимость одного выставленного бала из зарплаты. Для каждого учителя получаем его зарплату и сумму выставленных им оценок. Выводим таблицу с ФИО учителя, его ЗП, сумму выставленных баллов и стоимость одного балла полученную из ЗП/сумму баллов). 
WITH TeacherStats AS ( --что нам вообще надо
    SELECT 
        t.intTeacherId, --ид учителя
        t.intTeacherName, --имя
        t.fltTeacherSalary, -- зп
        SUM(m.intMarkValue) AS TotalMarksSum --считаем сумму всех выставленных баллов
    FROM tblTeacher t
    JOIN tblSubject s ON t.intTeacherId = s.intTeacherId
    JOIN tblLesson l ON s.intSubjectId = l.intSubjectId
    JOIN tblMark m ON l.intLessonId = m.intLessonId
    GROUP BY t.intTeacherId, t.intTeacherName, t.fltTeacherSalary
)

SELECT TOP 1
    intTeacherName,
    fltTeacherSalary,
    TotalMarksSum,
    CAST(fltTeacherSalary AS DECIMAL(10,2)) / CAST(TotalMarksSum AS DECIMAL(10,2)) AS CostPerPoint
FROM TeacherStats
ORDER BY CostPerPoint DESC;

--задание 5. Успеваемость. Получаем таблицу состоящую из ФИО ученика и его среднего балла по всем предметам
SELECT
    tblPupil.txtPupilSurname + ' ' + tblPupil.txtPupilName AS ФИО,
    AVG(tblMark.intMarkValue * 1.0) AS Средний_Балл
FROM tblPupil
JOIN tblMark ON tblPupil.intPupilId = tblMark.intPupilId
GROUP BY tblPupil.txtPupilSurname, tblPupil.txtPupilName
ORDER BY Средний_Балл ASC
LIMIT 10;

ХП - это некое подобие функции, которая хранится на сервере. ПЛюсы впечатляющие:
+ возможность запускать ХП без предоставления прав на БД
+ очевидно, все преимущества функций как в других ЯП - многократное переиспользование по имени
+ сокращение трафика для передачи запроса (ХП ведь лежит и ВЫПОЛНЯЕТСЯ на сервере. План выполнения сохраняется посе выполнения и запуск происходит мгновенно). КОмпиляция в P-Code (псевдокод)
+ расширение функционала через параметры вызова ХП
+ можно объединить несколько операций в одну транзакцию (группа операций которые ыдолжны выполниться вместе, или не выполниться не одной - атомарность)
* если много операций надо выполнить сразу пачкой, легко может случиться сбой на любом этапе и часть выполнится, а часть упадет - и как такое исправлять, это боль. А вот транзакция либо выполнится и всё ок, либо нет но ничего не сломается)

Схема работы:
1. Вызов: EXEC usp_CopyAllTables @CopyNumber = 1
2. Проверка: параметр @CopyNumber корректен? (не отрицательный, не NULL)
3. Поиск таблиц: найти все таблицы, начинающиеся на 'tbl' без цифр
4. Создать список: временная таблица #SourceTables
5. Для каждой таблицы в списке (помним про N'.'):
   5.1. Собрать имя: 'tblSubject' + '1' = 'tblSubject1'
   5.2. Удалить старую копию (если есть)
   5.3. Создать новую: SELECT * INTO tblSubject1 FROM tblSubject
   5.4. Напечатать в консоль

6. Очистить ресурсы (закрыть курсор, удалить временную таблицу)
7. Вывод в консоль "Копирование завершено"

9. Создайте хранимую процедуру, которая заполняет столбец intLessonCount («Количество проведенных уроков») в таблице tblSubject правильными данными. Проверьте работу процедуры.

Вспоминаем. Есть предмет intSubjectName в tbl.Subject, количество часов intSubjectVolume и количество уроков по предмету intLessonCnt

Итак, с чем можно столкнуться? Количество уроков intLessonCount в таблице tblSubject. Уроки tblLesson c FK intsubjectId.
!!! Поле intLessonCount может устареть (если добавлять или удалялять уроки напрямую, без обновления счетчика)

Чтобы этого избежать, надо чтобы хранимая процедура пересчитывала intLessonCount на основе свежих данных из tblLesson. Это самое главное

Как это работает
1. Первый UPDATE (обновляем предметы с уроками)
   Считаем уроки: SELECT intSubjectId, COUNT(*) FROM tblLesson GROUP BY intSubjectId
   Обновляем: UPDATE tblSubject SET intLessonCount = посчитанное_значение
2. Второй UPDATE (обнуляем предметы без уроков)
   Находим предметы, которых нет в tblLesson
   UPDATE tblSubject SET intLessonCount = 0
3. Выводим результаты (SELECT с LEFT JOIN)
4. Выводтим результат в консоль
